Configuration Set-up
--------------------
bin/muddrv.cfg              This is the master config-file, driver.pl wants this name as first param
bin/setup/actions.cfg       All standard actions coded in std/Actions.pm
bin/setup/adverbs.cfg       Adverb list
bin/setup/constants.cfg     Constant values you can access via getsetup('xxx')
bin/setup/emotes.cfg        All emote strings used by std/Actions.pm -> do_emote()
bin/setup/messages.cfg      Anywhere you want a standard message you can use  driver()->{Message}->{YourTag}

-----------------------------------------------------------------------------
run()
    while drv_is_alive
        ->linkdead_test
            foreach client in connections
                quit_client(.) if linkdead or idletime
        ->fulfil_callouts( curtime ) # for each expired pending callout,
            foreach tt ( ->callouts ) 
                current_user( that-player )
                delete ->callouts->{ tt };
                call_other( rest-params )
        foreach client ready, 
            alternatively calls...
            ->socket_logon
                new = ->listner->accept
                current_user( User->new )
                            Living->new
                                Object->new
                                    register_object
                            unregister_object
                cat( welcome )
            ->socket_process( client ) 
                pl = client_to_user( client ) || 0
                sysread client
                pl->input_number++
                pl->input_line = $` pl->input_buf = $'
                write_snoopees( $pl, $input_line )
                pl->input_to() 
                    alternatively:
                        ** process_logon()   
                        ** process_startup() 
                        ** process_normal()  
        ->heart_beat() # for each living-object (sets on current_user).
            foreach object in driver->objects
                current_user( object )
                object->heart_beat( curtime )
                on error showcomperr()
        sleep

process_startup()  
    room = current_user()->environment()

process_normal()  
    robot_check()
    alias-parser()
    split( /|;/ , input_line )
        force( input_line, pl ) # only Living can be forced
            command, param = split( input_line )
            do_command( command, param )            
                1. call_other( cmd/adm/_command, cmd_command", param ) if admin
                2. call_other( cmd/wiz/_command, cmd_command", param ) if wizard
                   call_other( cmd/file/_command, cmd_command", param ) if wizard
                3. foreach item in pl->inventory 
                     call_other( item, item->actions->{ command }, command, param );
                4. foreach item in pl->environment->inventory
                     call_other( item, item->actions->{ command }, command, param );
                5. call_other( room, room->actions->{ command }, command, param )
                #6. call_other( user, user->actions->{ command }, command, param )
                7. call_other( cmd/norm/_command, cmd_command", param )
                8. call_other( 'std/Actions', driver->actions->{ command }, command, param )
                9. call_other( cmd/ghost/_command, cmd_command", param )
        notify_fail() unless ok.

-----------------------------------------------------------------------------
call_other( obj, cmd, @_ )                --> -1:not match with errors; 0:not match.
    if  ref( obj )  
        obj->command( param )
    elsif  obj eq 'MudCore'     
        command(param) 
    else # !ref( obj )  
        pkg = load_module( obj )         
        pkg->new( param )
        pkg->command( param )     
        on error showcomperr


-----------------------------------------------------------------------------
load_module( $obj, $refresh )              
    unload_module( $obj ) if $refresh       --> -1:compilation errors; 0:object not found.
    include_file( $obj ".pl" )
